\begin{proof}
    In this proof, we'll utilise the following algebraic fact: For any ring \(R\) and any \(R\)-module \(M\), we have an isomorphism
    \[M\cong M^{\vee\vee}.\]
    In fact, this solves the problem over affine schemes. Applying this fact repeatedly, we've defined the morphisms over an open cover of our domain. Thus we only need to check if the morphism glues together. Over the intersection, they agree because localisation commutes with dual functor (i.e. the hom functor).
\end{proof}

\begin{proof}
    In this attempt, we'll use transition maps to define our $\mathscr E^{\vee}$. 
    Let $\{U_i\}$ be an open cover of $X$ such that $\psi_i:\mathscr E\vert_{U_i}\cong \mathcal O^{\oplus n}_{U_i}$.
    Then our transition maps are given by 
    \begin{align*}
        T_{ij}:= \psi_i\circ \psi_j^{-1}\in \GL_n(\mathcal O_X(U_i\cap U_j)).
    \end{align*}
    Now we need to examine the dual sheaf $\mathscr E^{\vee}$. Over $U_i$, we need a suitable choice for the isomorphism 
    \begin{align}
        \varphi_{i}:\mathscr E^{\vee}\vert_{U_i}\to \mathcal O^{\oplus n}_{U_i}\cong (\mathcal O^{\oplus n}_{U_i})^{\vee}.
    \end{align}And our choise will be $\varphi_{i}:=\psi_{i}^{\vee,-1}$.
    Therefore our new transition maps for $\mathscr E^{\vee}$ are given
    \begin{align}
        S_{ij}=&~\varphi_{i}\circ\varphi_{j}^{-1}\\
        =&~(\psi_{i}^{\vee,-1})\circ (\psi_{j}^{\vee,-1})^{-1}\\
        =&~(\psi_{i}^{\vee,-1})\circ \psi_{j}^{\vee}\\
        =&~(\psi_{j}\circ \psi_{i}^{-1})^{\vee}\\
        =&~T_{ij}^{-1,\vee}.
    \end{align}
    Thus applying twice the dual functor, the transition maps will restore the original $T_{ij}$, and we are done.
\end{proof}

\section{14.4.D.}
\begin{exercise}
    Suppose $\pi:X\to Y$ is a finite morphism of locally Noetherian schemes. If $\mathscr F$ is a coherent sheaf on $X$, show that $\pi_{\ast}\mathscr F$ is a coherent sheaf. Hint: Show first that $\pi_{\ast}\mathscr O_X$ is finite type.
\end{exercise}

\begin{proof}
    \begin{enumerate}
    \item In the following, we'll check $\pi_{\ast}\mathscr O_X$ is finite type.
    While being finite type satisfies the requirement of Affine Communication Lemma, it suffices to show the existence of a cover of affine opens $\{U_i\}_{i\in I}$ such that 
    \[\pi_{\ast}\mathscr O_X(U_i)\] is finitely generated. In fact, we may take $\{U_i\}_{i\in I}$ to be a cover of open affines inside $Y$. Recall the morphism $\pi$ is finite: in the textbook, it's automatically affine; hence we may assume $\pi^{-1}(U_i)=V_i$ where $V_i$ is affine. 

    According to the definition of finite morphism, we have the following exact sequence
    \begin{align}
        \mathscr O_Y(U_i)^{\oplus n} \to \pi_{\ast}\mathscr O_X(U_i)=\mathscr O_X(\pi^{-1}(U))=\mathscr O_X(V_i)\to 0.
    \end{align}for some choice of $n\in\mathbb Z$.
    So we may conclude $\pi_{\ast}\mathscr O_X$ is of finite type. 

    \item Since $\mathscr F$ is a coherent sheaf, under the locally Noetherian assumption, it is of finite type. Thus over $U_i$, the following is exact
    \begin{align}
        \mathscr O_X\vert_{U_i}^{\oplus l}\to \mathscr F\vert_{U_i}\to 0.
    \end{align}
    In general, the functor $\pi_{\ast}$ is only left-exact.
    One important fact is that $\pi$ is finite thus an affine morphism, so apply $\pi_{\ast}$ will be right-exact. Therefore we have the following exact.
    \begin{align}\label{14.4.D.1}
        \pi_{\ast}(\mathscr O_X\vert_{U_i}^{\oplus l})\to \pi_{\ast}\mathscr F\vert_{U_i}\to 0.
    \end{align}
    On the other hand, the last proof gave us 
    \begin{align}\label{14.4.D.2}
        \mathscr O_Y\vert_{U_i}^{\oplus m}\to \pi_{\ast}\mathscr O_X\vert_{U_i}\to 0.
    \end{align}
    Combining two exact sequences above, we may conclude over $U_i$, 
    \begin{align}
        (\mathscr O_Y\vert_{U_i})^{\oplus ml}\to \pi_{\ast}\mathscr F\vert_{U_i}\to 0
    \end{align} is exact. Therefore $\pi_{\ast}\mathscr F$ is of finite type.
    \end{enumerate}
\end{proof}
