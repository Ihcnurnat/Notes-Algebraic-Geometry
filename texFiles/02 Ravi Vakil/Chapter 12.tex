\section{12.1.B.}

\begin{proof}

    \begin{align*}
    \left\{
    \begin{array}{l}
    \text{irreducible closed subsets of } U
    \end{array}
    \right\}
    &\longrightarrow
    \left\{
    \begin{array}{l}
    \text{irreducible closed subsets of } X \\
    \text{that meet } U
    \end{array}
    \right\} \\[1em]
    %
    A &\mapsto \overline{A} \quad \text{(closure taken in } X\text{)} \\
    A \cap U &\mapsfrom A
    \end{align*}
\end{proof}
